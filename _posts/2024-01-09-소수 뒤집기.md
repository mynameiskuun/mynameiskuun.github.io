### 제출한 답안

```java
 public boolean isPrime(int num) {

        if(num==1) return false;
        for(int i=2; i<num; i++) {
             if(num % i == 0) {
                 return false;
             }
        }
        return true;
    }

    public List<Integer> solution(int count, int[] input) {

        List<Integer> answer = new ArrayList<>();

        for(int i=0; i<count; i++) {
            int tmp = input[i];
            int res = 0;
            while(tmp>0) {
                int t = tmp%10;
                res = res * 10 + t;
                tmp = tmp/10;
            }
            if(isPrime(res)) {
                answer.add(res);
            }
        }
        return answer;
    }
```
